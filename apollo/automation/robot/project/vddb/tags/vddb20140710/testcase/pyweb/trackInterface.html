comment #action:获取token #expect:返回token=XXXX
${token} = interface auth token
Comment #action:入库 dna ingest #expect:返回 code:303,且返回body含 meta_uuid
${meta_uuid}= pyweb ingest dna ${SAMPLE_DNA_VIDEO} ${token}
${inst_uuid} = get instance uuid from db ${meta_uuid}
Comment #action:调用 track_manager 接口 #expect:返回 code:200,且返回body含 instance_uuid
${track_no} Set Variable 0
${return} = Pyweb Api Call GET /mediawise/instances/${inst_uuid}/tracks/${track_no}/meta
... None {"Content-type":"application/x-www-form-urlencoded","Cookie": "${token}"}
check rsp_code ${return} ${200}
Comment #action:比较数据库的meta_uuid对应的 instance_uuid与返回的 instance_uuid是否一致 #exepect:结果一致
${instance_id} = Get Json Element ${return} body.body.instance_id
${video_duration} = Get Json Element ${return} body.body.video_duration
Should Be Equal ${inst_uuid} ${instance_id}
Comment #action:校验数据库 videoInstanceTrack对应的 Instance_uuid记录只有一条,且 track_number,video_duration与返回值一致 #exepect:结果与数据库一致
${result} = mysql_query select count(*) ,track_number,video_duration from videoInstanceTrack where instance_uuid='${inst_uuid}'
Should Be Equal '1' '${result[0][0]}'
Should Be Equal '${track_no}' '${result[0][1]}'
Should Be Equal '${video_duration}' '${result[0][2]}'
[Teardown] del meta and instance ${meta_uuid} ${inst_uuid}
